Q3. Find one employee from IT department.

ANSWER:	db.employees.findOne({ department: "IT" });



Q4. Display all employees.

ANSWER:	db.employees.find();



Q5. Display only name and salary.

ANSWER:	db.employees.find({}, { name: 1, salary: 1, _id: 0 });



Q6. Find employees with salary equal to 50000.

ANSWER:	db.employees.find({ salary: 50000 });



Q7. Find employees with salary less than 60000.

ANSWER:	db.employees.find({ salary: { $lt: 60000 } });



Q8. Find employees with salary greater than or equal to 60000.

ANSWER:	db.employees.find({ salary: { $gte: 60000 } });



Q9. Find employees whose age is not equal to 30.

ANSWER:	db.employees.find({ age: { $ne: 30 } });



Q10. Find employees from Chennai or Mumbai.

ANSWER:	db.employees.find({ city: { $in: ["Chennai", "Mumbai"] } });



Q11. Find employees not from Chennai.

ANSWER:	db.employees.find({ city: { $ne: "Chennai" } });



Q12. Find IT employees with salary > 50000.

ANSWER:	db.employees.find({ department: "IT", salary: { $gt: 50000 } });



Q13. Find employees from Chennai OR Bangalore.

ANSWER:	db.employees.find({ city: { $in: ["Chennai", "Bangalore"] } });



Q14. Find employees NOT earning above 60000.

ANSWER:	db.employees.find({ salary: { $not: { $gt: 60000 } } });



Q15. Find employees neither in IT nor HR.

ANSWER:	db.employees.find({ department: { $nin: ["IT", "HR"] } });



Q16. Find documents containing experience field.

ANSWER:	db.employees.find({ experience: { $exists: true } });



Q17. Find documents where salary is Integer.

ANSWER:	db.employees.find({ salary: { $type: "int" } });



Q18. Find employees having MongoDB skill.

ANSWER:	db.employees.find({ skills: "MongoDB" });



Q19. Find employees having both MongoDB and AWS.

ANSWER:	db.employees.find({ skills: { $all: ["MongoDB", "AWS"] } });



Q20. Find employees with exactly 3 skills.

ANSWER:	db.employees.find({ skills: { $size: 3 } });



Q21. Find employees having Java skill using elemMatch.

ANSWER:	db.employees.find({ skills: { $elemMatch: { $eq: "Java" } } });



Q22. Display employees sorted by salary descending.

ANSWER:	db.employees.find().sort({ salary: -1 });



Q23. Display top 3 highest paid employees.

ANSWER:	db.employees.find().sort({ salary: -1 }).limit(3);



Q24. Increase salary of employee 101 by 5000.

ANSWER:	db.employees.updateOne({ empId: 101 }, { $inc: { salary: 5000 } });



Q25. Increase salary of all IT employees by 10%.

ANSWER:	db.employees.updateMany({ department: "IT" }, { $mul: { salary: 1.10 } });
