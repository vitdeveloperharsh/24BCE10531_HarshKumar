1.
[JOINS] List the name of each employee along with their job title and department name.


ANSWER:  SELECT Employees.name, Jobs.job_title, Departments.dept_name
         FROM Employees
         INNER JOIN Jobs ON Employees.job_id = Jobs.job_id
         INNER JOIN Departments ON Employees.dept_id = Departments.dept_id;

2.
[SUBQUERY] Find all employees whose salary is higher than the average salary of the 'IT' department.


ANSWER:  SELECT * FROM Employees 
         WHERE salary > (
         SELECT AVG(salary) FROM Employees
         INNER JOIN Departments ON Employees.dept_id = Departments.dept_id
         WHERE Departments.dept_name = 'IT'
         );


3.
[GROUP BY] Calculate the total salary expenditure for each department name (not ID).



ANSWER:   SELECT Departments.dept_name, SUM(Employees.salary)
          FROM Departments
          INNER JOIN Employees ON Departments.dept_id = Employees.dept_id
          GROUP BY Departments.dept_name;


4.
[HAVING] Display departments that have more than 2 employees and an average salary above 70,000.



ANSWER:   SELECT Departments.dept_name
          FROM Departments
          INNER JOIN Employees ON Departments.dept_id = Employees.dept_id
          GROUP BY Departments.dept_name
          HAVING COUNT(Employees.emp_id) > 2 AND AVG(Employees.salary) > 70000;



5.
[SELF JOIN] Retrieve a list of employee names and their respective manager's names.



ANSWER:  SELECT Emp.name AS EmployeeName, Mgr.name AS ManagerName
         FROM Employees AS Emp
         LEFT JOIN Employees AS Mgr ON Emp.manager_id = Mgr.emp_id;


6.
[ORDER BY] Find the top 3 highest-paid employees and list them in descending order.



ANSWER:  SELECT name, salary FROM Employees
         ORDER BY salary DESC

9.
[HAVING] List job titles where the maximum salary of an employee in that role is less than 100,000.


ANSWER:   SELECT Jobs.job_title
          FROM Jobs
          INNER JOIN Employees ON Jobs.job_id = Employees.job_id
          GROUP BY Jobs.job_title
          HAVING MAX(Employees.salary) < 100000;

13.
[CORRELATED SUBQUERY] Find employees who earn more than the average salary of their own department.


ANSWER:     SELECT * FROM Employees AS MainEmp
            WHERE salary > (
                SELECT AVG(salary) FROM Employees AS SubEmp
                WHERE SubEmp.dept_id = MainEmp.dept_id
            );

16.
[SUBQUERY] List departments that do not have any 'Analysts'.


ANSWER:     SELECT dept_name FROM Departments
            WHERE dept_id NOT IN (
                SELECT dept_id FROM Employees
                INNER JOIN Jobs ON Employees.job_id = Jobs.job_id
                WHERE Jobs.job_title = 'Analyst'
            );



17.
[AGGREGATION] Find the difference between the highest and lowest salary in the company.



ANSWER:      SELECT MAX(salary) - MIN(salary) FROM Employees;



18.
[JOIN + ORDER BY] Display employee names and hire dates, sorted by department name (A-Z) and then by salary (High to Low).



ANSWER:      SELECT Departments.dept_name, Employees.name, Employees.hire_date, Employees.salary
             FROM Employees
             INNER JOIN Departments ON Employees.dept_id = Departments.dept_id
             ORDER BY Departments.dept_name ASC, Employees.salary DESC;



19.
[HAVING] Identify managers (by ID) who manage more than 2 people.


ANSWER:      SELECT manager_id
             FROM Employees
             WHERE manager_id IS NOT NULL
             GROUP BY manager_id
             HAVING COUNT(emp_id) > 2;


20.
[SUBQUERY] Find the employee(s) with the third-highest salary.



ANSWER:     SELECT * FROM Employees
            ORDER BY salary DESC
            LIMIT 1 OFFSET 2;


21.
[LEFT JOIN] List all job titles and the number of employees currently holding that job.



ANSWER:      SELECT Jobs.job_title, COUNT(Employees.emp_id)
             FROM Jobs
             LEFT JOIN Employees ON Jobs.job_id = Employees.job_id
             GROUP BY Jobs.job_title;

ANSWER:      SELECT Employees.name FROM Employees
             INNER JOIN Departments ON Employees.dept_id = Departments.dept_id
             WHERE Departments.dept_name = 'Finance' AND Employees.name LIKE '%i%';
