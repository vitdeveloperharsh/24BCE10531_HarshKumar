1.
List the Employee Name, their Department Name, and the Location Name for all employees.

ANSWER:   SELECT e.name AS employee_name, j.job_title, d.dept_name
          FROM Employees e
          JOIN Jobs j ON e.job_id = j.job_id
          JOIN Departments d ON e.dept_id = d.dept_id;



2.
Retrieve the Project Name, Client Name, and the Total Hours Worked on that project, but only for projects managed by the 'Consulting' department.


ANSWER:    SELECT Projects.project_name, Projects.client_name, SUM(Assignments.hours_worked) AS total_hours
           FROM Projects
           INNER JOIN Assignments ON Projects.project_id = Assignments.project_id
           INNER JOIN Departments ON Projects.dept_id = Departments.dept_id
           WHERE Departments.dept_name = 'Consulting'
           GROUP BY Projects.project_name, Projects.client_name;


3.
Find the names of Employees who have worked on a project with Client Name 'GlobalTech Solutions'. List the Employee Name and Project Name.


ANSWER:    SELECT Employees.employee_name, Projects.project_name
           FROM Employees
           INNER JOIN Assignments ON Employees.employee_id = Assignments.employee_id
           INNER JOIN Projects ON Assignments.project_id = Projects.project_id
           WHERE Projects.client_name = 'GlobalTech Solutions';

6.
Calculate the Average Hours Worked per Employee across all projects. List the Employee Name and the average hours.


ANSWER:		SELECT Employees.employee_name, AVG(Assignments.hours_worked)
		FROM Employees, Assignments
		WHERE Employees.employee_id = Assignments.employee_id
		GROUP BY Employees.employee_name;	


7.
Find the Department Name(s) that have an average salary above the company-wide average salary.


ANSWER:		SELECT Departments.dept_name
		FROM Departments, Employees
		WHERE Departments.dept_id = Employees.dept_id
		GROUP BY Departments.dept_name
		HAVING AVG(Employees.salary) > (SELECT AVG(salary) FROM Employees);



8.
Identify the Client Name(s) whose projects have a total combined hours worked exceeding 500 hours.


ANSWER:		SELECT Projects.client_name
		FROM Projects, Assignments
		WHERE Projects.project_id = Assignments.project_id
		GROUP BY Projects.client_name
		HAVING SUM(Assignments.hours_worked) > 500;





9.
For each Location, find the Department Name with the highest count of employees.


ANSWER:    SELECT Departments.location_id, Departments.dept_name
           FROM Departments, Employees
           WHERE Departments.dept_id = Employees.dept_id
	   GROUP BY Departments.location_id, Departments.dept_name
	   ORDER BY COUNT(Employees.employee_id) DESC
	   LIMIT 1;



10.
Calculate the total number of employees assigned to each Project, but only include projects that have more than 5 employees assigned.


ANSWER:  SELECT Projects.project_name, COUNT(Assignments.employee_id)
         FROM Projects, Assignments
         WHERE Projects.project_id = Assignments.project_id
         GROUP BY Projects.project_name
         HAVING COUNT(Assignments.employee_id) > 5;

12.
Retrieve the Department Name(s) that have employees assigned to more than one project.


ANSWER:		SELECT Departments.dept_name
		FROM Departments, Employees, Assignments
		WHERE Departments.dept_id = Employees.dept_id
  		AND Employees.employee_id = Assignments.employee_id
		GROUP BY Departments.dept_name
		HAVING COUNT(DISTINCT Assignments.project_id) > 1;


13.
Write a query to list the Employee Name and Project Name for any employee who has worked less than 10 hours on a project.



ANSWER:		SELECT Employees.employee_name, Projects.project_name
		FROM Employees, Assignments, Projects
		WHERE Employees.employee_id = Assignments.employee_id
  		AND Assignments.project_id = Projects.project_id
  		AND Assignments.hours_worked < 10;

15.
Select the Employee Name(s) whose salary is greater than the maximum salary in their own department.


ANSWER:   SELECT employee_name FROM Employees AS E1
          WHERE salary >= (
                 SELECT MAX(salary) FROM Employees AS E2
                 WHERE E2.dept_id = E1.dept_id
                   );


16.
Assuming the Assignments table has a start_date column, find the Employee Name and their oldest project start date.


ANSWER:   SELECT Employees.employee_name, MIN(Assignments.start_date)
          FROM Employees, Assignments
          WHERE Employees.employee_id = Assignments.employee_id
          GROUP BY Employees.employee_name;

19.
Find the top 3 Departments based on the total hours worked by all their employees across all projects.


ANSWER:     SELECT Departments.dept_name, SUM(Assignments.hours_worked)
           FROM Departments, Employees, Assignments
            WHERE Departments.dept_id = Employees.dept_id
                AND Employees.employee_id = Assignments.employee_id
            GROUP BY Departments.dept_name
            ORDER BY SUM(Assignments.hours_worked) DESC
            LIMIT 3;


  
20.
Display the Project Name and a status ("High Activity" if total hours > 300, "Normal" otherwise) using the CASE statement.


ANSWER:    SELECT Projects.project_name,
           CASE 
                WHEN SUM(Assignments.hours_worked) > 300 THEN 'High Activity'
                ELSE 'Normal'
           END
           FROM Projects, Assignments
           WHERE Projects.project_id = Assignments.project_id
           GROUP BY Projects.project_name;


21.
List the Employee Name and their Manager's Name, even if the employee has no manager, or if a potential manager has no employees reporting to them.



ANSWER:    SELECT Workers.employee_name, Managers.employee_name
           FROM Employees AS Workers
           LEFT JOIN Employees AS Managers ON Workers.manager_id = Managers.employee_id;
